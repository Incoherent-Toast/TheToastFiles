// Nametag check
func IsNametagged (Weapon, WeaponID)
  ?"" ! Weapon
    var Weapon_Split = string.Split(Weapon)
    var isLost = false
    for i = 1..3
      ?Weapon_Split[Weapon_Split.Count()-i] = "lost"
        isLost = true
        break
    ?isLost
      ?WeaponID = "cult_mask"
        WeaponID = "cultist_mask"
      :?WeaponID = "blade_of_god"
        WeaponID = "blade_of_the_fallen_god"
      ?string.Sub(WeaponID, 0, 2) = "ae"
        WeaponID = "æ" + string.Sub(WeaponID, 2, string.Size(WeaponID) - 2)
      
      var WeaponID_Split = string.Split(WeaponID, "_", true)
      var weaponNameTagArr = []
      var name_index = 0
      for i = 0..Weapon_Split.Count()-1
        ?i - name_index = WeaponID_Split.Count()
          break
        ?!string.Equals(WeaponID_Split[i - name_index], string.ToLower(Weapon_Split[i]))
          name_index ++
          weaponNameTagArr.Add(Weapon_Split[i])
      ?name_index = 0
        return ""
      return string.Join(" ", weaponNameTagArr)
    
    var WeaponID_Split = string.Split(WeaponID, ":", true)
    var WeaponID_Split_2 = []
    WeaponID_Split_2 = string.Split(WeaponID_Split[WeaponID_Split.Count()-1], "_", true)
    
    ?WeaponID_Split_2.Count() > 1
      for a : WeaponID_Split_2  
        ?string.IndexOf(Weapon, a) = 0 & !string.Equals(Weapon_Split[3], "runestone")
          return a
  return ""

// Colors by enchantment tier 
var clr
func itemColor (ench, item_id)
  ?ench & ench ! " "
    ?ench >= 16
      clr = "#rainFF"
    :?ench >= 11
      clr = "#red"
    :?ench >= 7
      clr = "#blue"
    :?ench >= 4
      clr = "#green"
    :?ench >= 2
      clr = "#yellow"
    :?ench = 1
      clr = "#cyan"
    :
      clr = "[Something went wrong]"
  :
    var idlist = ["star_stone", "ouroboros_stone", "fissure_stone", "triskelion_stone", "grappling_hook"]
    clr = "#ffffff"
    for id_ : idlist
      ?item_id = id_
        var lv_list = ["lv4", "lv3", "lv2"]
        var clr_list = ["#green", "#yellow", "#cyan"]
        for lv : lv_list
          ?item_id = lv
            clr = clr_list[lv_list.IndexOf(lv)]
            break
        break
  return clr

    ?Xarr1[RUNEi] = "stone" | Xarr1[RUNEi] = "gravity" | Xarr1[RUNEi] = "acrobatic"

        break

      runeX = "none"
      break
  return runeX

func DetermineItem (dataX)
  ?dataX.Count() & [nametagR, nametagL].Contains(dataX[0])
    dataX.RemoveAt(0)

func ToggleASCII(x, y, value, only)

  ASCIIL.visible = value
  ?only = true
    ASCIIPanelR.visible = false
    ASCIIPanelL.visible = false
    TextPanelR.visible = false
    TextPanelL.visible = false
    TextR.visible = false
    TextL.visible = false
  ?value = true

    ASCIIPanelL.y = y
    ASCIIPanelR.y = y

    ASCIIPanelL.visible = false
    TextPanelR.visible = false
    TextPanelL.visible = false
    for i = 0..comp_list.Count()-1
      comp_list[i].visible = vis_list[i]
